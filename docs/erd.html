    %% ── STORE & PLATFORM ──
    platforms {
        int id PK
        varchar name UK
        varchar display_name
    }

    stores {
        int id PK
        uuid uuid UK
        varchar name UK
        varchar display_name
        varchar base_url
        varchar scraper_type
        jsonb scraper_config
        int platform_id FK
        varchar platform_type
        jsonb discovery_config
        boolean is_active
    }

    mtg_singles_collections {
        int id PK
        varchar slug UK
        varchar display_name
    }

    %% ── SCRYFALL REFERENCE ──
    sets {
        int id PK
        varchar code UK
        varchar name
    }

    card_printings {
        int id PK
        int card_name_id FK
        uuid scryfall_id UK
        int set_id FK
        varchar collector_number
        varchar rarity
        text image_uri
        text layout
    }

    %% ── CARD IDENTITY (single table) ──
    card_names {
        int id PK
        varchar name
        varchar normalized_name UK
        uuid oracle_id "nullable - from Scryfall"
    }

    %% ── DISCOVERY ──
    product_urls {
        bigint id PK
        int store_id FK
        int collection_id FK
        varchar handle
        varchar validation_status
        varchar extraction_status
        text extraction_error
        int variants_total
    }

    %% ── CENTRAL HUB ──
    card_listings {
        bigint id PK
        int card_name_id FK "ALWAYS SET"
        int card_printing_id FK "NULLABLE"
        int store_id FK
        bigint product_url_id FK
        varchar title
        varchar set_name
        varchar set_code
        varchar condition
        boolean foil
        decimal price
        varchar currency
        boolean in_stock
        int quantity
        text product_link
        varchar platform_variant_id
    }

    %% ── PRICE HISTORY ──
    card_price_history {
        bigint id PK
        bigint card_listing_id FK
        int card_printing_id FK
        int store_id FK
        decimal price
        varchar condition
        boolean foil
        timestamp recorded_at
    }

    %% ── RELATIONSHIPS ──
    platforms ||--o{ stores : "platform_id"
    stores ||--o{ product_urls : "store_id"
    mtg_singles_collections ||--o{ product_urls : "collection_id"
    product_urls ||--o{ card_listings : "product_url_id"
    stores ||--o{ card_listings : "store_id"

    card_names ||--o{ card_printings : "card_name_id"
    sets ||--o{ card_printings : "set_id"

    card_names ||--o{ card_listings : "card_name_id"
    card_printings |o--o{ card_listings : "card_printing_id"

    card_listings ||--o{ card_price_history : "card_listing_id"
    card_printings |o--o{ card_price_history : "card_printing_id"
    stores ||--o{ card_price_history : "store_id"
